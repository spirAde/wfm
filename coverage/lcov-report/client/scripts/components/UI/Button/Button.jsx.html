import React, { PropTypes } from 'react';
import Link from 'react-router/lib/Link';
import classNames from 'classnames/bind';
 
import Icon from '../Icon/Icon';
import Spinner from '../Spinner/Spinner';
 
import styles from './Button.css';
 
const cx = classNames.bind(styles);
 
const Button = ({
  kind,
  isDisabled,
  isLoading,
  icon,
  iconAlign,
  label,
  to,
  className,
  onClick,
  style,
  buttonRef,
}) => {
  const classes = cx(className, styles.Button, {
    Disabled: isDisabled || isLoading,
    Primary: kind === 'primary',
    Secondary: kind === 'secondary',
    Back: kind === 'back',
  });
 
  let renderedComponent = null;
 
  if (kind === 'back') {
    return (
      <Link className={classes} to={to} onClick={!to && onClick ? onClick : Function.prototype}>
        <div className={styles.Inner}>
          <Icon icon="arrow-left" width={16} height={16} style={{ marginRight: 5 }} />
          <div className={styles.Label}>{label}</div>
        </div>
      </Link>
    );
  } else if (isLoading) {
    renderedComponent = (
      <div className={styles.Inner}>
        <div className={styles.Label}>{label}</div>
        <Spinner size="xs" color="#a29bb5" center style={{ marginLeft: 5 }} />
      </div>
    );
  } else {
    renderedComponent = (
      <div className={styles.Inner}>
        {
          icon && iconAlign === 'left' ?
            <Icon icon={icon} width={20} height={20} style={{ marginRight: 5 }} /> :
            null
        }
        <div className={styles.Label}>{label}</div>
        {
          icon && iconAlign === 'right' ?
            <Icon icon={icon} width={20} height={20} style={{ marginLeft: 5 }} /> :
            null
        }
      </div>
    );
  }
 
  return (
    <button
      className={classes}
      style={style}
      disabled={isDisabled || isLoading}
      onClick={isDisabled || isLoading ? () => {} : onClick}
      ref={buttonRef}
    >
      {renderedComponent}
    </button>
  );
};
 
Button.displayName = 'Button';
 
Button.propTypes = {
  kind: PropTypes.oneOf(['primary', 'secondary', 'back']),
  isDisabled: PropTypes.bool,
  isLoading: PropTypes.bool,
  icon: PropTypes.string,
  iconAlign: PropTypes.oneOf(['left', 'right']),
  to: PropTypes.string,
  className: PropTypes.string,
  onClick: PropTypes.func,
  label: PropTypes.string,
  style: PropTypes.object,
  buttonRef: PropTypes.func,
};
 
Button.defaultProps = {
  kind: 'primary',
  isDisabled: false,
  isLoading: false,
  icon: '',
  iconAlign: 'right',
  to: '',
  className: '',
  onClick: Function.prototype,
  label: '',
  style: {},
  buttonRef: Function.prototype,
};
 
export default Button;
