import React, { Component, PropTypes } from 'react';
import classNames from 'classnames';
 
import styles from './RadioGroup.css';
 
const cx = classNames.bind(styles);
 
class RadioGroup extends Component {
  constructor(props) {
    super(props);
 
    this.state = {
      checkedValue: props.initialChecked,
    };
 
    this.handleChildChange = this.handleChildChange.bind(this);
  }
 
  handleChildChange(value, checked) {
    const { checkedValue } = this.state;
 
    if (checkedValue !== value && checked) {
      this.setState({ checkedValue: value });
      this.props.onChange(value);
    }
  }
 
  renderChildren() {
    const { checkedValue } = this.state;
    const { children } = this.props;
 
    return children.map((child, idx) => {
      const { checked, defaultChecked, name } = child.props;
 
      const isChecked = checked || defaultChecked || checkedValue === name;
 
      return React.cloneElement(child, {
        onChange: this.handleChildChange,
        key: `radio-${idx}`,
        checked: isChecked,
        className: styles.Item,
      });
    });
  }
 
  render() {
    const { align } = this.props;
 
    const classes = cx(styles.RadioGroup, {
      [styles.AlignRow]: align === 'row',
    });
 
    const renderedChildren = this.renderChildren();
 
    return (
      <div className={classes}>
        {renderedChildren}
      </div>
    );
  }
}
 
RadioGroup.displayName = 'RadioGroup';
 
RadioGroup.propTypes = {
  align: PropTypes.oneOf(['row', 'column']),
  onChange: PropTypes.func,
  initialChecked: PropTypes.string,
  children: PropTypes.node,
};
 
RadioGroup.defaultProps = {
  onChange: Function.prototype,
  initialChecked: '',
  align: 'column',
  children: [],
};
 
export default RadioGroup;
