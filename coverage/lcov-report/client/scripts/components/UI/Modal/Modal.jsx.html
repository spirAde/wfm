import React, { PropTypes } from 'react';
import OverlayModal from 'react-overlays/lib/Modal';
 
import assign from 'lodash/assign';
 
import Icon from '../Icon/Icon';
 
const Modal = ({
  show,
  isClosable,
  children,
  style,
  backdropStyle,
  backdropClassName,
  onHide,
  ...props
}) => (
  <OverlayModal
    show={show}
    backdropStyle={
      assign({}, backdropStyle, Modal.defaultStyles.backdrop, style.backdrop || {})
    }
    backdropClassName={backdropClassName}
    onBackdropClick={onHide}
    onEscapeKeyUp={onHide}
    {...props}
  >
    <div
      style={assign({}, Modal.defaultStyles.content, style.content || {})}
    >
      {
        isClosable &&
        <div style={Modal.defaultStyles.closeZone} onClick={onHide}>
          <Icon icon="close" style={{ opacity: 0.4, stroke: 'none' }} />
        </div>
      }
      {children}
    </div>
  </OverlayModal>
);
 
Modal.displayName = 'Modal';
 
Modal.propTypes = {
  ...OverlayModal.PropTypes,
  isClosable: PropTypes.bool,
  children: PropTypes.node,
  style: PropTypes.shape({
    backdrop: PropTypes.object,
    content: PropTypes.object,
    closeZone: PropTypes.object,
  }),
};
 
Modal.defaultProps = {
  show: false,
  isClosable: true,
  style: {
    backdrop: {},
    content: {},
    closeZone: {},
  },
  children: [],
};
 
Modal.defaultStyles = {
  backdrop: {
    position: 'fixed',
    top: 0,
    left: 0,
    right: 0,
    bottom: 0,
    backgroundColor: 'rgba(0, 0, 0, 0.7)',
    zIndex: 10,
  },
  content: {
    maxHeight: 'calc(100% - 100px)',
    position: 'fixed',
    top: '50%',
    left: '50%',
    transform: 'translate(-50%, -50%)',
    border: '1px solid #ccc',
    background: '#fff',
    WebkitOverflowScrolling: 'touch',
    borderRadius: 5,
    outline: 'none',
    padding: '20px 26px',
    width: 614,
    zIndex: 11,
  },
  closeZone: {
    position: 'absolute',
    top: 5,
    right: 5,
  },
};
 
export default Modal;
