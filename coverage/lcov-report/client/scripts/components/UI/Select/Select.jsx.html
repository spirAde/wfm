import React, { PropTypes, Component } from 'react';
import classNames from 'classnames/bind';
import onClickOutside from 'react-onclickoutside';
 
import find from 'lodash/find';
 
import Icon from '../Icon/Icon';
import Option from './Option/Option';
 
import styles from './Select.css';
 
const cx = classNames.bind(styles);
 
class Select extends Component {
  constructor(props) {
    super(props);
 
    this.state = {
      selectedValue: props.selectedValue,
      isOpen: false,
    };
 
    this.handleClickToggleMenu = this.handleClickToggleMenu.bind(this);
    this.handleChange = this.handleChange.bind(this);
  }
 
  handleClickOutside() {
    this.setState({
      isOpen: false,
    });
  }
 
  handleChange(value) {
    this.setState({
      selectedValue: value,
    }, () => this.props.onChange(value));
  }
 
  handleClickToggleMenu(event) {
    event.preventDefault();
 
    const { isOpen } = this.state;
 
    this.setState({
      isOpen: !isOpen,
    });
  }
 
  renderBox() {
    const { isOpen } = this.state;
 
    if (!isOpen) return null;
 
    const renderedOptions = this.renderOptions();
 
    return (
      <div className={styles.Box}>
        <div className={styles.OptionList}>
          {renderedOptions}
        </div>
      </div>
    );
  }
 
  renderOptions() {
    const { options } = this.props;
    const { selectedValue } = this.state;
 
    return options.map((option, index) => {
      const { value } = option;
 
      const isSelected = value === selectedValue;
 
      return (
        <Option
          isSelected={isSelected}
          option={option}
          key={index}
          onClick={this.handleChange}
        />
      );
    });
  }
 
  render() {
    const { selectedValue, isOpen } = this.state;
    const { options, placeholder, errorText, onBlur, errorClassName } = this.props;
 
    const label = selectedValue
      ? find(options, option => option.value === selectedValue).label || placeholder
      : placeholder;
 
    const isError = !!errorText;
    const renderedBox = this.renderBox();
 
    const fieldClasses = cx(styles.Field, {
      Error: isError,
    });
 
    const labelClasses = cx(styles.Label, {
      Placeholder: !selectedValue && placeholder,
    });
 
    const errorClasses = cx(styles.ErrorField, errorClassName);
 
    return (
      <div className={styles.Select} onClick={this.handleClickToggleMenu} tabIndex="0" onBlur={onBlur}>
        <div className={fieldClasses}>
          <div className={labelClasses}>
            <span>{label}</span>
          </div>
          <span className={styles.ArrowZone}>
            {
              isOpen
              ? <Icon icon="arrow-up" stroke="#464646" />
              : <Icon icon="arrow-down" stroke="#464646" />
            }
          </span>
        </div>
        {renderedBox}
        {isError && <div className={errorClasses}>{errorText}</div>}
      </div>
    );
  }
}
 
Select.displayName = 'Select';
 
Select.propTypes = {
  selectedValue: PropTypes.oneOfType([
    PropTypes.string,
    PropTypes.number,
    PropTypes.bool,
  ]),
  options: PropTypes.arrayOf(
    PropTypes.object,
  ).isRequired,
  placeholder: PropTypes.string,
  errorText: PropTypes.oneOfType([
    PropTypes.string,
    PropTypes.bool,
  ]),
  errorClassName: PropTypes.string,
  onChange: PropTypes.func,
  onBlur: PropTypes.func,
};
 
Select.defaultProps = {
  placeholder: '',
  selectedValue: undefined,
  options: [],
  errorClassName: '',
  errorText: '',
  onChange: Function.prototype,
  onBlur: Function.prototype,
};
 
export default onClickOutside(Select);
export { Select as PureSelect }; // for test
